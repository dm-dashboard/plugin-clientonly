Client Only Plugin

This plugin has no server side config. Configure each widget as you add it to the dashboard.

iFrame

{
    "url" : "http://pathToUrl/" //Required
}

Youtube

{
    //Unique ID used by Youtube
    "videoId" : "flkwejfiv",
    //Url to video on YouTube. Either the ID or the Url is required. If both are supplied, the ID is used
    "videoUrl" : "http://pathToUrl/",

    "loop" : true|false, //Optional, defaults to false
    "autoplay" : true|false //Optional, defaults to true
}

Image

{
    "url" : "pathToAnImage",
    "stretchBoth":true|false, //Stretch to fill screen, aspect ratio will not be respected
    "stretchWidth":true|false, //Stretch width to fill screen, aspect ratio will be respected
    "stretchHeight":true|false, //Stretch height to fill screen, aspect ratio will be respected
}
